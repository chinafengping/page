page的demo

以下是分页的demo，具体说api请点击。

完全版分页

new Pagination({
container: "#page1", // 容器
total_page: 11, // 总页数
continuous_page: 5, // 连续显示分页数
current_page: 6, // 当前页
first_page: rue, // 首页
preview_show: true, // 是否显示第一页
next_show: true, // 是否显示下一页
last_show: true, // 最后一页
edge_page: 2, // 两端显示分页数
skip_show: true // 是否显示跳转
});

淘宝版分页

new Pagination({ container: "#page2", // 容器
total_page: 20, // 总页数
continuous_page: 5, // 连续显示分页数
current_page: 7, // 当前页
preview_show: true, // 是否显示第一页
next_show: true, // 是否显示下一页
edge_page: 0 // 两端显示分页数
});

上页下页

new Pagination({
container: "#page3", // 容器
total_page: 0, // 总页数
next_show: true, // 是否显示下一页
preview_show: true // 是否显示上一页
});
